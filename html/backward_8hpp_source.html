661  _Unwind_Reason_Code backtrace(_Unwind_Context* ctx) {
662  if (_index >= 0 && static_cast<size_t>(_index) >= _depth)
663  return _URC_END_OF_STACK;
664 
665  int ip_before_instruction = 0;
666  uintptr_t ip = _Unwind_GetIPInfo(ctx, &ip_before_instruction);
667 
668  if (!ip_before_instruction) {

675  }
676

678  (*_f)(_index, (void*)ip);
